# ── NORMALIZE ──────────────────────────────────────────────────────
def normalize(value, window):
    if len(window) < 5: return 0.0
    mean = sum(window) / len(window)
    variance = sum((x-mean)**2 for x in window) / len(window)
    std = variance**0.5
    if std == 0: return 0.0
    return max(-10.0, min(10.0, ((value-mean)/std)*3.0))

def make_shape_id(v):
    h = hashlib.sha256(json.dumps([round(x,1) for x in v]).encode()).hexdigest()[:8].upper()
    return f"SHP_{h}"

def make_sequence_id(sh):
    h = hashlib.sha256(json.dumps(list(sh)).encode()).hexdigest()[:8].upper()
    return f"SEQ_{h}"

def state_distance(vec, nom):
    if nom is None: return 0.0
    keys = ["nominal_inertia","nominal_tension","nominal_compression",
            "nominal_expansion","nominal_turbulence","nominal_memory"]
    try:
        diffs = [(vec[i] - float(nom[keys[i]]))**2 for i in range(6)]
        return round(math.sqrt(sum(diffs) / 6), 3)
    except: return 0.0

# ── TOPOLOGY DETECTION ─────────────────────────────────────────────
def detect_topology(inertia, tension, compression, expansion,
                    turbulence, energy, prev_e, regime_dur):
    if tension > 6.5 and (energy - prev_e) > 1.8:      return "rupture"
    if prev_e > 4.0 and energy < 1.5:                   return "collapse"
    if expansion > 6.0 and prev_e < 2.0 and energy > 3: return "bloom"
    if compression > 5.0 and regime_dur > 8 and tension < 1.0: return "fold"
    if energy < 0.4 and regime_dur > 20:                return "eerie_coherent"
    return None

# ── NOMINAL COMPUTE ────────────────────────────────────────────────
def compute_nominal(timescale):
    try:
        limit_map = {"24h":5760,"3d":17280,"7d":40320}
        limit = limit_map.get(timescale, 5760)
        result = supabase.table("states")\
            .select("inertia,tension,compression,expansion,turbulence,memory,energy")\
            .eq("asset", ASSET).lt("energy", 1.5)\
            .order("epoch", desc=True).limit(limit).execute()
        rows = result.data
        if len(rows) < 10:
            print(f"  Nominal {timescale}: insufficient data ({len(rows)} rows)")
            return
        keys = ["inertia","tension","compression","expansion","turbulence","memory"]
        nom = {f"nominal_{k}": round(sum(float(r[k]) for r in rows)/len(rows),3) for k in keys}
        nom.update({"asset":ASSET,"timescale":timescale,"computed_at_epoch":market["epoch"],
                    "confidence":min(1.0,len(rows)/1000),"low_energy_samples":len(rows)})
        supabase.table("nominal_states").upsert(nom, on_conflict="asset,timescale").execute()
        nominal[timescale] = nom
        print(f"  Nominal {timescale} updated — {len(rows)} samples")
    except Exception as e:
        print(f"  Nominal {timescale} error: {e}")

network_cache = {"data":None,"ts":0}

def compute_rarity(shape_id, total):
    try:
        res = supabase.table("pattern_index")\
            .select("occurrence_count").eq("shape_id",shape_id).eq("asset",ASSET)\
            .limit(1).execute()
        if res.data:
            occ = int(res.data[0]["occurrence_count"])
            return round(1.0-(occ/max(1,total)),3)
    except: pass
    return 0.95

# ── PHYSICS ────────────────────────────────────────────────────────
def compute_state():
    global prev_energy,prev_regime,regime_start,total_epochs,prev_shape_id

    p=market["price"]; b=market["bid"]; a=market["ask"]
    bvol=market["buy_vol"]; svol=market["sell_vol"]; ep=market["epoch"]

    deriv   = fetch_derivatives()
    fund    = deriv["funding"]; oi_value=deriv["oi_value"]
    macro   = fetch_macro()
    eth_dom = macro["eth_dominance"]; btc_dom=macro["btc_dominance"]; mcap_chg=macro["mcap_change"]
    network = fetch_network()
    gas     = network["gas_price"]; base_fee=network["base_fee"]
    defi    = fetch_defi()
    tvl_vel = defi["tvl_velocity"]; dex_vol=defi["dex_vol"]
    staking = fetch_staking()
    staking_apr = staking["staking_apr"]
    social  = fetch_social()
    fng=social["fng"]; trends=social["trends"]; cg_sent=social["cg_sentiment"]
    cc_social=social["cc_social"]; news_vel=social["news_velocity"]
    cg_price=social["cg_price"]; cg_change=social["cg_change"]
    eth_btc=social["eth_btc_ratio"]

    # Windows
    if p>0: price_window.append(p)
    spread=(a-b)/b if b>0 else 0
    spread_window.append(spread)
    flow=bvol-svol; flow_window.append(flow)
    if bvol+svol>0: volume_window.append(bvol+svol)
    funding_window.append(fund)
    if oi_value>0: oi_window.append(oi_value)
    dom_window.append(eth_dom); mcap_window.append(mcap_chg)
    gas_window.append(gas); fee_window.append(base_fee)
    tvl_window.append(tvl_vel)
    if dex_vol>0: dex_vol_window.append(dex_vol)
    fng_window.append(fng); trends_window.append(trends)
    cg_sent_window.append(cg_sent); cc_social_window.append(cc_social)
    news_vel_window.append(news_vel)
    if cg_price>0: cg_price_window.append(cg_price)
    cg_spread_window.append(cg_change)
    if eth_btc>0: eth_btc_window.append(eth_btc)
    staking_window.append(staking_apr)

    # Normalize
    price_sig    = normalize(p, price_window)
    spread_sig   = normalize(spread, spread_window)
    flow_sig     = normalize(flow, flow_window)
    vol_sig      = normalize(bvol+svol, volume_window)
    fund_sig     = normalize(fund, funding_window)
    oi_sig       = normalize(oi_value, oi_window)
    eth_dom_sig  = normalize(eth_dom, dom_window)
    mcap_sig     = normalize(mcap_chg, mcap_window)
    # ETH/BTC ratio as dominance signal — rising ETH/BTC = risk-on expansion
    eth_btc_sig  = normalize(eth_btc, eth_btc_window) if len(eth_btc_window)>4 else 0
    macro_sig    = round((eth_dom_sig + mcap_sig + eth_btc_sig*0.5)/2.5, 3)
    # Gas = network activity — high gas = expansion, low = compression
    gas_sig      = normalize(gas, gas_window)
    fee_sig      = normalize(base_fee, fee_window)
    network_sig  = round((gas_sig + fee_sig)/2, 3)
    tvl_sig      = normalize(tvl_vel, tvl_window)
    dex_sig      = normalize(dex_vol, dex_vol_window)
    defi_sig     = round((tvl_sig+dex_sig)/2, 3)
    staking_sig  = normalize(staking_apr, staking_window)
    fng_signed   = (fng-50)/5
    trends_sig   = normalize(trends, trends_window)
    cg_sig       = normalize(cg_sent, cg_sent_window)
    cc_sig       = normalize(cc_social, cc_social_window)
    news_sig     = normalize(news_vel, news_vel_window)
    cg_price_sig = normalize(cg_price, cg_price_window) if len(cg_price_window)>4 else price_sig
    cg_change_sig= normalize(cg_change, cg_spread_window) if len(cg_spread_window)>4 else 0
    price_divergence = round(abs(price_sig-cg_price_sig), 3)

    social_sig = max(-10.0, min(10.0, round(
        fng_signed*0.30 + trends_sig*0.20 + cg_sig*0.15 +
        cc_sig*0.20 + news_sig*0.10 + eth_btc_sig*0.05, 3)))

    # ETH physics — gas feeds inertia + expansion, staking feeds memory
    inertia     = round(price_sig*0.35 + fund_sig*0.25 + oi_sig*0.20 + macro_sig*0.10 + gas_sig*0.10, 3)
    tension     = round(spread_sig*0.30 + abs(flow_sig)*0.20 + abs(fund_sig)*0.18 + abs(defi_sig)*0.12 + price_divergence*0.20, 3)
    compression = round(-vol_sig*0.30 + spread_sig*0.30 + -oi_sig*0.15 + -gas_sig*0.15 + staking_sig*0.10, 3)
    expansion   = round(vol_sig*0.25 + flow_sig*0.18 + oi_sig*0.12 + defi_sig*0.15 + social_sig*0.12 + gas_sig*0.10 + cg_change_sig*0.08, 3)
    turbulence  = round(abs(flow_sig)*0.25 + spread_sig*0.25 + abs(fund_sig)*0.15 + abs(gas_sig)*0.15 + abs(social_sig)*0.20, 3)
    memory      = round((inertia+expansion+staking_sig*0.2)/2.2, 3)
    energy      = round((inertia**2+tension**2+compression**2+expansion**2+turbulence**2+memory**2)**0.5/6**0.5, 3)
    energy_delta = round(energy-prev_energy, 3)

    # Regime
    if energy<1.5:      regime="nominal"
    elif expansion>4:   regime="expanding_coherent"
    elif turbulence>5:  regime="turbulent"
    elif compression>4: regime="compressing"
    elif tension>5:     regime="high_tension"
    else:               regime="active"

    regime_changed = regime!=prev_regime
    if regime_changed: regime_start=ep
    regime_duration = ep-regime_start+1

    topology_event = detect_topology(
        inertia,tension,compression,expansion,
        turbulence,energy,prev_energy,regime_duration)

    state_vec   = [inertia,tension,compression,expansion,turbulence,memory]
    shape_id    = make_shape_id(state_vec)
    shape_history.append(shape_id)
    sequence_id = make_sequence_id(shape_history)

    dev_24h = state_distance(state_vec, nominal["24h"])
    dev_3d  = state_distance(state_vec, nominal["3d"])
    dev_7d  = state_distance(state_vec, nominal["7d"])

    total_epochs_local = total_epochs if total_epochs>0 else 1
    rarity = compute_rarity(shape_id, total_epochs_local)

    state = {
        "asset":ASSET,"timestamp":int(time.time()*1000),"epoch":ep,
        "inertia":inertia,"tension":tension,"compression":compression,
        "expansion":expansion,"turbulence":turbulence,"memory":memory,
        "energy":energy,"energy_delta":energy_delta,
        "regime":regime,"regime_duration":regime_duration,"regime_changed":regime_changed,
        "topology_event":topology_event,
        "shape_id":shape_id,"prev_shape_id":prev_shape_id,"sequence_id":sequence_id,
        "raw_price":round(p,2),"fear_greed":fng,"btc_dominance":round(btc_dom,2),
        "dev_from_24h":dev_24h,"dev_from_3d":dev_3d,"dev_from_7d":dev_7d,
        "rarity_score":rarity,
        "price_signal":round(price_sig,3),"micro_signal":round(flow_sig,3),
        "deriv_signal":round((fund_sig+oi_sig)/2,3),
        "chain_signal":round((network_sig+defi_sig)/2,3),
        "social_signal":social_sig,"macro_signal":macro_sig,
    }

    prev_energy=energy; prev_regime=regime; prev_shape_id=shape_id; total_epochs+=1
    return state

# ── WRITE ──────────────────────────────────────────────────────────
def write_state(state):
    global prev_state_id
    ep=state["epoch"]
    try:
        result   = supabase.table("states").insert(state).execute()
        state_id = result.data[0]["id"]
        print(f"✓ {ep:5d} | {state['shape_id']} | "
              f"I:{state['inertia']:+.2f} T:{state['tension']:+.2f} "
              f"E:{state['energy']:.2f} Δ:{state['energy_delta']:+.2f} | "
              f"{state['regime']}"
              + (f" ⚡{state['topology_event']}" if state['topology_event'] else ""))
        if prev_state_id is not None:
            supabase.table("transitions").insert({
                "asset":ASSET,"from_state_id":prev_state_id,"to_state_id":state_id,
                "from_shape_id":state["prev_shape_id"] or state["shape_id"],
                "to_shape_id":state["shape_id"],"sequence_id":state["sequence_id"],
                "elapsed_epochs":1,"topology_event":state["topology_event"],
            }).execute()
        if state["topology_event"]:
            supabase.table("topology_events").insert({
                "asset":ASSET,"event_type":state["topology_event"],"state_id":state_id,
                "epoch":ep,"energy_at_event":state["energy"],
                "prev_energy":state["energy"]-state["energy_delta"],
                "regime_before":state["regime"],"regime_after":state["regime"],
                "inertia":state["inertia"],"tension":state["tension"],
                "compression":state["compression"],"expansion":state["expansion"],
                "turbulence":state["turbulence"],"timestamp":state["timestamp"],
            }).execute()
            print(f"  ⚡ ETH Topology: {state['topology_event']}")
        supabase.rpc("increment_pattern_index",{
            "p_shape_id":state["shape_id"],"p_asset":ASSET,
            "p_energy":state["energy"],"p_epoch":ep,"p_rarity":state["rarity_score"],
        }).execute()
        supabase.table("sequences").upsert({
            "asset":ASSET,"sequence_id":state["sequence_id"],
            "shape_ids":json.dumps(list(shape_history)),
            "occurrence_count":1,"last_seen_epoch":ep,"avg_energy":state["energy"],
        },on_conflict="sequence_id,asset").execute()
        supabase.table("meta").upsert({
            "key":f"total_epochs_{ASSET}","value":str(total_epochs),
        },on_conflict="key").execute()
        prev_state_id=state_id
        if ep%5760==0:  compute_nominal("24h")
        if ep%17280==0: compute_nominal("3d")
        if ep%40320==0: compute_nominal("7d")
    except Exception as e:
        print(f"✗ Write error: {e}")

# ── BYBIT STREAM ───────────────────────────────────────────────────
async def bybit_stream():
    uri = "wss://stream.bybit.com/v5/public/linear"
    print("Connecting to Bybit (ETH)...")
    async with websockets.connect(uri) as ws:
        await ws.send(json.dumps({"op":"subscribe","args":["orderbook.1.ETHUSDT","publicTrade.ETHUSDT"]}))
        print("Subscribed to ETH streams")
        print("─"*60)
        epoch_start=time.time(); market["epoch"]=1
        async for msg in ws:
            data=json.loads(msg)
            if "topic" not in data: continue
            topic=data["topic"]
            if topic.startswith("orderbook"):
                try:
                    ob=data["data"]
                    if ob.get("b") and ob.get("a"):
